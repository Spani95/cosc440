MODULE_NAME = asgn2
EXTRA_CFLAGS += -Werror


obj-m   := $(MODULE_NAME).o
#asgn2-objs := gpio.o

KDIR    := /lib/modules/$(shell uname -r)/build
PWD     := $(shell pwd)



all: module

module:
	$(MAKE) -C $(KDIR) M=$(PWD) modules


clean:
	$(MAKE) -C $(KDIR) M=$(PWD) clean

help:
	$(MAKE) -C $(KDIR) M=$(PWD) help

install:
	$(MAKE) -C $(KDIR) M=$(PWD) modules_install
